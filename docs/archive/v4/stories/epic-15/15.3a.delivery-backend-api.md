# Story 15.3a: Backend Delivery API

## Status

Done

## Story

**As a** разработчик платформы FREESPORT,
**I want** создать backend API для управления способами доставки,
**so that** frontend может получить список доступных способов доставки, а администратор может управлять ими через Django Admin.

## Acceptance Criteria

1. ✅ Создан Django app `apps/delivery/` с моделью `DeliveryMethod` и конфигурацией `DeliveryConfig`
2. ✅ API endpoint `GET /api/v1/delivery/methods/` возвращает список способов доставки
3. ✅ Модель содержит поля: id, name, description, icon, is_available, sort_order
4. ✅ Django Admin интерфейс для управления способами доставки
5. ✅ Fixtures с начальными данными (Курьер, Самовывоз, Транспортная компания)
6. ✅ Unit-тесты для API endpoint (включая edge cases: retrieve single, filter by is_available)
7. ✅ OpenAPI документация (drf-spectacular)

### Задача 2: Модель DeliveryMethod (AC: 1, 3)

- [x] Создать модель в `apps/delivery/models.py`:

  ```python
  from django.db import models

  class DeliveryMethod(models.Model):
      """Способ доставки"""

      id = models.CharField(max_length=50, primary_key=True)  # courier, pickup, transport
      name = models.CharField("Название", max_length=100)
      description = models.TextField("Описание", blank=True)
      icon = models.CharField("Иконка", max_length=50, blank=True, help_text="CSS класс иконки или emoji для UI")
      is_available = models.BooleanField("Доступен", default=True)
      sort_order = models.PositiveIntegerField("Порядок сортировки", default=0)
      created_at = models.DateTimeField(auto_now_add=True)
      updated_at = models.DateTimeField(auto_now=True)

      class Meta:
          verbose_name = "Способ доставки"
          verbose_name_plural = "Способы доставки"
          ordering = ["sort_order", "name"]

      def __str__(self):
          return self.name
  ```

- [x] Создать `apps/delivery/views.py`:

  ```python
  from drf_spectacular.utils import extend_schema, extend_schema_view
  from rest_framework import viewsets, permissions

  from .models import DeliveryMethod
  from .serializers import DeliveryMethodSerializer

  @extend_schema_view(
      list=extend_schema(
          summary="Список способов доставки",
          description="Возвращает список доступных способов доставки",
          tags=["Delivery"]
      ),
      retrieve=extend_schema(
          summary="Получить способ доставки",
          description="Возвращает детали конкретного способа доставки по ID",
          tags=["Delivery"]
      )
  )
  class DeliveryMethodViewSet(viewsets.ReadOnlyModelViewSet):
      """
      ViewSet для получения списка способов доставки.
      Только чтение - управление через Django Admin.
      """
      queryset = DeliveryMethod.objects.filter(is_available=True)
      serializer_class = DeliveryMethodSerializer
      permission_classes = [permissions.AllowAny]  # Доступно без авторизации
  ```

### Задача 7: Unit-тесты API (AC: 6)

- [x] Создать `apps/delivery/tests/test_delivery_api.py`:

  ```python
  import pytest
  from rest_framework.test import APIClient
  from apps.delivery.models import DeliveryMethod

  @pytest.fixture
  def api_client():
      return APIClient()

  @pytest.fixture
  def delivery_methods():
      return [
          DeliveryMethod.objects.create(
              id="courier", name="Курьер", description="До двери",
              icon="🚚", is_available=True, sort_order=1
          ),
          DeliveryMethod.objects.create(
              id="pickup", name="Самовывоз", description="Забрать",
              icon="🏪", is_available=True, sort_order=2
          ),
          DeliveryMethod.objects.create(
              id="disabled", name="Отключён", description="",
              icon="❌", is_available=False, sort_order=3
          ),
      ]

  @pytest.mark.django_db
  class TestDeliveryMethodsListAPI:
      """Тесты для GET /api/v1/delivery/methods/"""

      def test_list_returns_available_methods(self, api_client, delivery_methods):
          """Проверяет, что API возвращает только доступные способы доставки"""
          response = api_client.get("/api/v1/delivery/methods/")
          assert response.status_code == 200
          assert len(response.data) == 2  # Только is_available=True

      def test_list_excludes_unavailable(self, api_client, delivery_methods):
          """Проверяет, что недоступные способы исключаются из списка"""
          response = api_client.get("/api/v1/delivery/methods/")
          ids = [m["id"] for m in response.data]
          assert "disabled" not in ids

      def test_response_structure(self, api_client, delivery_methods):
          """Проверяет структуру ответа API"""
          response = api_client.get("/api/v1/delivery/methods/")
          method = response.data[0]
          assert "id" in method
          assert "name" in method
          assert "description" in method
          assert "icon" in method
          assert "is_available" in method

      def test_empty_list_returns_200(self, api_client):
          """Проверяет, что пустой список возвращает 200"""
          response = api_client.get("/api/v1/delivery/methods/")
          assert response.status_code == 200
          assert response.data == []

      def test_ordering_by_sort_order(self, api_client, delivery_methods):
          """Проверяет сортировку по sort_order"""
          response = api_client.get("/api/v1/delivery/methods/")
          ids = [m["id"] for m in response.data]
          assert ids == ["courier", "pickup"]  # Сортировка по sort_order

  @pytest.mark.django_db
  class TestDeliveryMethodRetrieveAPI:
      """Тесты для GET /api/v1/delivery/methods/{id}/"""

      def test_retrieve_single_method(self, api_client, delivery_methods):
          """Проверяет получение одного способа доставки по ID"""
          response = api_client.get("/api/v1/delivery/methods/courier/")
          assert response.status_code == 200
          assert response.data["id"] == "courier"
          assert response.data["name"] == "Курьер"
          assert response.data["icon"] == "🚚"

      def test_retrieve_unavailable_method_returns_404(self, api_client, delivery_methods):
          """Проверяет, что недоступный способ возвращает 404"""
          response = api_client.get("/api/v1/delivery/methods/disabled/")
          assert response.status_code == 404

      def test_retrieve_nonexistent_method_returns_404(self, api_client):
          """Проверяет, что несуществующий ID возвращает 404"""
          response = api_client.get("/api/v1/delivery/methods/nonexistent/")
          assert response.status_code == 404
  ```
