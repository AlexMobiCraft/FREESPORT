**ДОБАВИТЬ новую секцию: Badge Logic**

````markdown
### Badge Rendering Logic

**Источник:** Product model поля из Story 11.0

**Логика отображения бейджа (приоритет):**

```typescript
// src/components/common/ProductBadge.tsx
import type { Product } from '@/types/api';
import { Badge } from '@/components/ui/Badge';

interface ProductBadgeProps {
  product: Product;
}

export function ProductBadge({ product }: ProductBadgeProps) {
  // Приоритет 1: Скидка
  if (product.is_sale && product.discount_percent) {
    return <Badge variant="sale">{product.discount_percent}% скидка</Badge>;
  }

  // Приоритет 2: Акция
  if (product.is_promo) {
    return <Badge variant="promo">Акция</Badge>;
  }

  // Приоритет 3: Новинка
  if (product.is_new) {
    return <Badge variant="new">Новинка</Badge>;
  }

  // Приоритет 4: Хит продаж
  if (product.is_hit) {
    return <Badge variant="hit">Хит продаж</Badge>;
  }

  // Приоритет 5: Премиум
  if (product.is_premium) {
    return <Badge variant="premium">Premium</Badge>;
  }

  return null; // Нет бейджа
}
```
````
