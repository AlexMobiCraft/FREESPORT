  - [ ] 4.5.1 Создать `frontend/src/hooks/useDebounce.ts`:

    ```typescript
    import { useState, useEffect } from "react";

    export function useDebounce<T>(value: T, delay: number): T {
      const [debouncedValue, setDebouncedValue] = useState<T>(value);

      useEffect(() => {
        const timer = setTimeout(() => setDebouncedValue(value), delay);
        return () => clearTimeout(timer);
      }, [value, delay]);

      return debouncedValue;
    }
    ```

### Ключевые цвета Design System v2.0 [Source: docs/front-end-spec.md, docs/frontend/design-system.json]

```scss
// Primary (основной цвет для кнопок и checked states)
$primary-default: #1f1f1f; // Для кнопок, checkbox checked
$primary-hover: #333333;

// Interactive/Accent (для интерактивных элементов)
$accent-interactive: #0060ff; // Toggle on, focus rings, selected chips, slider active
$accent-hover: #0050cc;

// Neutral
$neutral-border: #d0d7e6; // Borders default
$neutral-bg: #e3e8f2; // Toggle off, slider track
$checkbox-border: #b9c3d6; // Checkbox border default

// Text
$text-primary: #1b1b1b;
$text-secondary: #4b5c7a; // Chip default text
$text-muted: #7a7a7a; // Icons, placeholders

// Surface
$surface-white: #ffffff;
$surface-hover: #f5f7fa; // Option hover
$surface-selected: #e1f0ff; // Selected option bg
```

**Примечание:** Checkbox использует `$primary-default` (#1F1F1F) для checked state, а Toggle, Chip selected, PriceRangeSlider и focus rings используют `$accent-interactive` (#0060FF).

// Select.test.tsx
import { render, screen, fireEvent } from '@testing-library/react'
import userEvent from '@testing-library/user-event'
import { Select } from '../Select'

const options = [
  { value: 'opt1', label: 'Option 1' },
  { value: 'opt2', label: 'Option 2' },
  { value: 'opt3', label: 'Option 3' }
]

describe('Select', () => {
  it('opens dropdown on click', async () => {
    render(<Select options={options} onChange={() => {}} />)

    await userEvent.click(screen.getByRole('combobox'))
    expect(screen.getByRole('listbox')).toBeInTheDocument()
  })

  it('selects option on click', async () => {
    const handleChange = vi.fn()
    render(<Select options={options} onChange={handleChange} />)

    await userEvent.click(screen.getByRole('combobox'))
    await userEvent.click(screen.getByText('Option 2'))

    expect(handleChange).toHaveBeenCalledWith('opt2')
  })

  it('navigates with arrow keys', async () => {
    render(<Select options={options} onChange={() => {}} />)

    const trigger = screen.getByRole('combobox')
    trigger.focus()
    await userEvent.keyboard('{ArrowDown}')

    expect(screen.getByRole('listbox')).toBeInTheDocument()

    await userEvent.keyboard('{ArrowDown}')
    await userEvent.keyboard('{Enter}')

    expect(screen.queryByRole('listbox')).not.toBeInTheDocument()
  })

  it('closes on Escape', async () => {
    render(<Select options={options} onChange={() => {}} />)

    await userEvent.click(screen.getByRole('combobox'))
    await userEvent.keyboard('{Escape}')

    expect(screen.queryByRole('listbox')).not.toBeInTheDocument()
  })
})
